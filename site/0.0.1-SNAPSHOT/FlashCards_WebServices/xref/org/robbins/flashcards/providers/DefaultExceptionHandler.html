2   package org.robbins.flashcards.providers;
3   
4   import javax.ws.rs.WebApplicationException;
5   import javax.ws.rs.core.Response;
6   import javax.ws.rs.core.Response.ResponseBuilder;
7   import javax.ws.rs.ext.ExceptionMapper;
8   import javax.ws.rs.ext.Provider;
9   
10  import org.robbins.flashcards.webservices.exceptions.GenericWebServiceException;
11  import org.springframework.stereotype.Component;
12  
13  @Provider
14  @Component
15  public class DefaultExceptionHandler implements ExceptionMapper<WebApplicationException> {
16  
17      @Override
18      public Response toResponse(final WebApplicationException webApplicationException) {
19          ResponseBuilder builder;
20          GenericWebServiceException exception;
21  
22          if (webApplicationException instanceof GenericWebServiceException) {
23              exception = (GenericWebServiceException) webApplicationException;
24          } else {
25              exception = new GenericWebServiceException(
26                      Response.Status.fromStatusCode(webApplicationException.getResponse().getStatus()),
27                      webApplicationException.getMessage());
28          }
29  
30          builder = Response.status(exception.getError().getErrorId());
31          builder.entity(exception.getError());
32          return builder.build();
33      }
34  }
