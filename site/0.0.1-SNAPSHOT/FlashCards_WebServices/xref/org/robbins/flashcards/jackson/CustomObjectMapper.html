2   package org.robbins.flashcards.jackson;
3   
4   import com.fasterxml.jackson.databind.ObjectMapper;
5   import com.fasterxml.jackson.databind.SerializationFeature;
6   import com.fasterxml.jackson.databind.ser.impl.SimpleFilterProvider;
7   import com.fasterxml.jackson.datatype.hibernate4.Hibernate4Module;
8   import com.fasterxml.jackson.datatype.hibernate4.Hibernate4Module.Feature;
9   import com.fasterxml.jackson.datatype.joda.JodaModule;
10  
11  public class CustomObjectMapper extends ObjectMapper {
12  
13      public CustomObjectMapper() {
14          Hibernate4Module hm = new Hibernate4Module();
15          hm.disable(Feature.FORCE_LAZY_LOADING);
16          this.registerModule(hm);
17  
18          this.disable(SerializationFeature.FAIL_ON_EMPTY_BEANS);
19          this.registerModule(new JodaModule());
20  
21          SimpleFilterProvider filterProvider = new SimpleFilterProvider();
22          filterProvider.setFailOnUnknownId(false);
23          this.setFilters(filterProvider);
24      }
25  
26      public void setPrettyPrint(final boolean prettyPrint) {
27          configure(SerializationFeature.INDENT_OUTPUT, prettyPrint);
28      }
29  }
