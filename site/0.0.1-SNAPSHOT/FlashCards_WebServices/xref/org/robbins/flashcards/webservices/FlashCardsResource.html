34  @Path("/v1/flashcards/")
35  @Component("flashCardsResource")
36  @Api(value = "/v1/flashcards", description = "Operations about FlashCards")
37  @Produces({ "application/xml", "application/json" })
38  @Consumes({ "application/xml", "application/json" })
39  public class FlashCardsResource extends AbstractGenericResource<FlashCardDto, Long> {
40  
41      @Inject
42      private FlashcardFacade flashcardFacade;
43  
44      @Override
45      protected GenericCrudFacade<FlashCardDto> getFacade() {
46          return flashcardFacade;
47      }
48  
49      @GET
50      @Path("/search")
51      @ApiOperation(value = "Search for FlashCards", response = FlashCardDto.class)
52      public FlashCardDto[] search(@QueryParam("page") final Integer page,
53              @DefaultValue("25") @QueryParam("size") final Integer size,
54              @QueryParam("question") final String question,
55              @QueryParam("tags") final String tags) {
56  
57          return searchFlashCards(page, size, question, tags);
58      }
59  
60      @GET
61      @Path("/search/count")
62      @ApiOperation(value = "Get a count of FlashCards", response = Long.class)
63      public Long searchCount(@QueryParam("question") final String question,
64              @QueryParam("tags") final String tags) {
65  
66          return Long.valueOf(searchFlashCards(null, null, question, tags).length);
67      }
68  
69      private FlashCardDto[] searchFlashCards(final Integer page, final Integer size,
70              final String question, final String tags) {
71  
72          try {
73              // find by Question
74              if (!StringUtils.isBlank(question)) {
75                  // are we using pagination?
76                  if (page != null) {
77                      List<FlashCardDto> results;
78                      results = flashcardFacade.findByQuestionLike(question,
79                              new PageRequest(page, size));
80                      return results.toArray(new FlashCardDto[results.size()]);
81                  } else {
82                      List<FlashCardDto> results = flashcardFacade.findByQuestionLike(question);
83                      return results.toArray(new FlashCardDto[results.size()]);
84                  }
85              }
86              // find by Tags
87              if (!StringUtils.isBlank(tags)) {
88                  StringTokenizer st = new StringTokenizer(tags, ",");
89                  Set<TagDto> tagsSet = new HashSet<TagDto>();
90                  while (st.hasMoreTokens()) {
91                      tagsSet.add(new TagDto(Long.parseLong(st.nextToken())));
92                  }
93                  // are we using Pagination?
94                  if (page != null) {
95                      List<FlashCardDto> results = flashcardFacade.findByTagsIn(tagsSet,
96                              new PageRequest(page, size));
97                      return results.toArray(new FlashCardDto[results.size()]);
98                  } else {
99                      List<FlashCardDto> results = flashcardFacade.findByTagsIn(tagsSet);
100                     return results.toArray(new FlashCardDto[results.size()]);
101                 }
102             } else {
103                 throw new GenericWebServiceException(Response.Status.BAD_REQUEST,
104                         "Invalid 'search' parameters");
105             }
106         } catch (ServiceException e) {
107             throw new GenericWebServiceException(Response.Status.INTERNAL_SERVER_ERROR, e);
108         }
109     }
110 
111     @Override
112     @PUT
113     @Path("/{id}")
114     public Response put(@PathParam("id") final Long id, final FlashCardDto dto) {
115 
116         // some client apps don't know the Created By and Created Date, so make
117         // sure we set it
118         if (dto.getCreatedBy() == null) {
119             FlashCardDto orig;
120             try {
121                 orig = flashcardFacade.findOne(id, null);
122             } catch (ServiceException e) {
123                 throw new GenericWebServiceException(
124                         Response.Status.INTERNAL_SERVER_ERROR, e);
125             }
126             dto.setCreatedBy(orig.getCreatedBy());
127             dto.setCreatedDate(orig.getCreatedDate());
128         }
129 
130         return super.put(id, dto);
131     }
