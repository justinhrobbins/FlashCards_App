2   package org.robbins.flashcards.facade.impl;
3   
4   import java.util.ArrayList;
5   import java.util.List;
6   import java.util.Set;
7   
8   import javax.inject.Inject;
9   
10  import org.robbins.flashcards.dto.TagDto;
11  import org.robbins.flashcards.exceptions.ServiceException;
12  import org.robbins.flashcards.facade.TagFacade;
13  import org.robbins.flashcards.facade.base.AbstractCrudFacadeImpl;
14  import org.robbins.flashcards.model.Tag;
15  import org.robbins.flashcards.service.TagService;
16  import org.robbins.flashcards.service.util.DtoUtil;
17  import org.springframework.stereotype.Component;
18  
19  @Component
20  public class DefaultTagFacade extends AbstractCrudFacadeImpl<TagDto, Tag> implements
21          TagFacade {
22  
23      @Inject
24      private TagService service;
25  
26      @Override
27      public TagService getService() {
28          return service;
29      }
30  
31      @Override
32      public TagDto findByName(final String name) {
33          Tag result = service.findByName(name);
34  
35          if (result == null) {
36              return null;
37          }
38          TagDto tagDto = getMapper().map(result, TagDto.class);
39          return tagDto;
40      }
41  
42      @Override
43      public TagDto getDto(final Tag entity) throws ServiceException {
44          return getDto(entity, null);
45      }
46  
47      @Override
48      public TagDto getDto(final Tag entity, final Set<String> fields)
49              throws ServiceException {
50          TagDto tagDto = getMapper().map(entity, TagDto.class);
51          DtoUtil.filterFields(tagDto, fields);
52          return tagDto;
53      }
54  
55      @Override
56      public Tag getEntity(final TagDto dto) {
57          return getMapper().map(dto, Tag.class);
58      }
59  
60      @Override
61      public List<TagDto> getDtos(final List<Tag> entities, final Set<String> fields)
62              throws ServiceException {
63          List<TagDto> dtos = new ArrayList<TagDto>();
64          for (Tag entity : entities) {
65              dtos.add(getDto(entity, fields));
66          }
67          return dtos;
68      }
69  
70      @Override
71      public List<Tag> getEtnties(final List<TagDto> dtos) {
72          List<Tag> entities = new ArrayList<Tag>();
73          for (TagDto dto : dtos) {
74              entities.add(getEntity(dto));
75          }
76          return entities;
77      }
78  }
