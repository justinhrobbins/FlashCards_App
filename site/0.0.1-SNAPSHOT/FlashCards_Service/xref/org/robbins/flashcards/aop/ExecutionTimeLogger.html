1   package org.robbins.flashcards.aop;
2   
3   import org.aspectj.lang.ProceedingJoinPoint;
4   import org.aspectj.lang.annotation.Around;
5   import org.aspectj.lang.annotation.Aspect;
6   import org.slf4j.Logger;
7   import org.slf4j.LoggerFactory;
8   import org.springframework.util.StopWatch;
9   
10  @Aspect
11  public class ExecutionTimeLogger {
12  	static final Logger logger = LoggerFactory.getLogger(ExecutionTimeLogger.class);
13  
14  	@Around("execution(* org.robbins.flashcards..*.*(..))")
15  	public Object logTimeMethod(ProceedingJoinPoint joinPoint) throws Throwable {
16  
17  		final StopWatch stopWatch = new StopWatch();
18  		stopWatch.start();
19  
20  		final Object retVal = joinPoint.proceed();
21  
22  		stopWatch.stop();
23  
24  		final StringBuffer logMessage = new StringBuffer();
25  		logMessage
26  				.append(joinPoint.getSignature().getDeclaringType().getName());
27  		logMessage.append(".");
28  		logMessage.append(joinPoint.getSignature().getName());
29  
30  		logMessage.append("(");
31  
32  		// append args
33  		Object[] args = joinPoint.getArgs();
34  		for (int i = 0; i < args.length; i++) {
35  			logMessage.append(args[i]).append(",");
36  		}
37  		if (args.length > 0) {
38  			logMessage.deleteCharAt(logMessage.length() - 1);
39  		}
40  
41  		logMessage.append(")");
42  
43  		logMessage.append(" execution time: ");
44  		logMessage.append(stopWatch.getTotalTimeMillis());
45  		logMessage.append(" ms");
46  		logger.debug(logMessage.toString());
47  		return retVal;
48  	}
49  }
