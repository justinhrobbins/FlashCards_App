2   package org.robbins.flashcards.presentation.converter;
3   
4   import java.util.Map;
5   
6   import org.apache.struts2.util.StrutsTypeConverter;
7   
8   import com.opensymphony.xwork2.conversion.TypeConversionException;
9   
10  public class IdConverter extends StrutsTypeConverter {
11  
12      @Override
13      public Object convertFromString(@SuppressWarnings("rawtypes") Map context,
14              String[] values, @SuppressWarnings("rawtypes") Class toClass) {
15          if ((values != null) && (values.length > 0) && (!values[0].equals("null"))) {
16              try {
17                  return Long.parseLong(values[0]);
18              } catch (NumberFormatException e) {
19                  throw new TypeConversionException(e);
20              }
21          }
22          return null;
23      }
24  
25      @Override
26      public String convertToString(@SuppressWarnings("rawtypes") Map context, Object o) {
27          return String.valueOf(o);
28      }
29  }
