3   import java.io.BufferedInputStream;
4   import java.io.BufferedOutputStream;
5   import java.io.IOException;
6   import java.io.InputStream;
7   import java.io.OutputStream;
8   import java.net.HttpURLConnection;
9   import java.net.MalformedURLException;
10  import java.net.URL;
11  import java.util.Enumeration;
12  import java.util.HashSet;
13  
14  import javax.servlet.ServletConfig;
15  import javax.servlet.ServletException;
16  import javax.servlet.http.HttpServlet;
17  import javax.servlet.http.HttpServletRequest;
18  import javax.servlet.http.HttpServletResponse;
19  
20  public class ProxyServlet extends HttpServlet {
21  
22  	private String hostname = null;
23  	private int port = -1;
24  	private String path = null;
25  	private static HashSet<String> ignoreResponseHeaders = new HashSet<String>();
26  	private static final long serialVersionUID = -1794977331184160392L;
27  	public static final String METHOD_GET = "GET";
28  	public static final String METHOD_POST = "POST";
29  	public static final String METHOD_PUT = "PUT";
30  	public static final String METHOD_DELETE = "DELETE";
31  	public static final String HOSTNAME_PARAM = "hostname";
32  	public static final String PORT_PARAM = "port";
33  	public static final String PATH_PARAM = "path";
34  	public static final int BUFFER_SIZE = 1024;
35  
36  	static {
37  		ignoreResponseHeaders.add("Transfer-Encoding");
38  	}
39  
40  	@Override
41  	public void init(ServletConfig config) throws ServletException {
42  
43  		super.init();
44  
45  		hostname = config.getInitParameter(HOSTNAME_PARAM);
46  		if (hostname == null) {
47  			throw new ServletException("Hostname is required.");
48  		}
49  
50  		String portHeader = config.getInitParameter(PORT_PARAM);
51  		port = (portHeader == null) ? -1 : Integer.parseInt(portHeader);
52  
53  		path = config.getInitParameter(PATH_PARAM);
54  		if (path == null) {
55  			throw new ServletException("Path is required.");
56  		}
57  	}
58  
59  	@Override
60  	public void destroy() {
61  
62  	}
63  
64  	@Override
65  	protected void service(HttpServletRequest request, HttpServletResponse response) throws IOException, ServletException {
66  
67  		// Construct the URL
68  		String path = this.path;
69  		String pathInfo = request.getPathInfo();
70  		if (pathInfo != null) {
71  			path += pathInfo;
72  		}
73  
74  		String queryString = request.getQueryString();
75  		if (queryString != null) {
76  			path += "?" + queryString;
77  		}
78  
79  		URL url = null;
80  		try {
81  			url = new URL(request.getScheme(), hostname, port, path);
82  		} catch (MalformedURLException exception) {
83  			throw new ServletException("Unable to construct URL.", exception);
84  		}
85  
86  		String method = request.getMethod();
87  
88  		// Open a connection to the URL
89  		HttpURLConnection connection = (HttpURLConnection) url.openConnection();
90  		connection.setRequestMethod(method);
91  		connection.setAllowUserInteraction(false);
92  		connection.setInstanceFollowRedirects(false);
93  		connection.setUseCaches(false);
94  
95  		// Write request headers to connection
96  		Enumeration<String> headerNames = request.getHeaderNames();
97  		if (headerNames != null) {
98  			while (headerNames.hasMoreElements()) {
99  				String headerName = headerNames.nextElement();
100 				Enumeration<String> headerValues = request.getHeaders(headerName);
101 
102 				while (headerValues.hasMoreElements()) {
103 					String headerValue = headerValues.nextElement();
104 					if (connection.getRequestProperty(headerName) == null) {
105 						connection.setRequestProperty(headerName, headerValue);
106 					} else {
107 						connection.addRequestProperty(headerName, headerValue);
108 					}
109 				}
110 			}
111 		}
112 
113 		// Set the input/output state
114 		connection.setDoInput(true);
115 		connection.setDoOutput(method.equalsIgnoreCase(METHOD_POST) || method.equalsIgnoreCase(METHOD_PUT));
116 
117 		// Connect to the server
118 		connection.connect();
119 
120 		// Write the request body
121 		if (connection.getDoOutput()) {
122 			OutputStream outputStream = null;
123 			try {
124 				InputStream inputStream = request.getInputStream();
125 				outputStream = connection.getOutputStream();
126 				outputStream = new BufferedOutputStream(outputStream, BUFFER_SIZE);
127 
128 				for (int data = inputStream.read(); data != -1; data = inputStream.read()) {
129 					outputStream.write((byte) data);
130 				}
131 
132 			} finally {
133 				if (outputStream != null) {
134 					outputStream.close();
135 				}
136 			}
137 		}
138 
139 		// Set the response status
140 		int status = connection.getResponseCode();
141 		int statusPrefix = status / 100;
142 		if (statusPrefix == 1 || statusPrefix == 3) {
143 			throw new ServletException("Unexpected server response: " + status);
144 		}
145 
146 		response.setStatus(status);
147 
148 		// Write response headers
149 		// NOTE Header indexes start at 1, not 0
150 		int i = 1;
151 		for (String key = connection.getHeaderFieldKey(i);
152 		key != null;
153 		key = connection.getHeaderFieldKey(++i)) {
154 			if (!ignoreResponseHeaders.contains(key)) {
155 				String value = connection.getHeaderField(i);
156 				if (response.containsHeader(key)) {
157 					response.addHeader(key, value);
158 				} else {
159 					response.setHeader(key, value);
160 				}
161 			}
162 		}
163 
164 		// Read the response body
165 		if (method.equalsIgnoreCase(METHOD_GET) || method.equalsIgnoreCase(METHOD_POST)) {
166 			InputStream inputStream = null;
167 			try {
168 				try {
169 					// Response returned on input stream
170 					inputStream = connection.getInputStream();
171 				} catch (Exception exception) {
172 					// Response returned on error stream
173 					inputStream = connection.getErrorStream();
174 				}
175 				if (inputStream != null) {
176 					inputStream = new BufferedInputStream(inputStream, BUFFER_SIZE);
177 					OutputStream outputStream = response.getOutputStream();
178 					for (int data = inputStream.read(); data != -1; data = inputStream.read()) {
179 						outputStream.write((byte) data);
180 					}
181 				}
182 				response.flushBuffer();
183 			} finally {
184 				if (inputStream != null) {
185 					inputStream.close();
186 				}
187 			}
188 		}
189 	}
190 }
