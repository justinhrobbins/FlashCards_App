16  import com.google.gwt.core.client.GWT;
17  import com.google.gwt.event.shared.EventBus;
18  import com.google.gwt.user.client.Window;
19  import com.google.gwt.user.client.ui.AcceptsOneWidget;
20  
21  public class ShellActivity extends AppAbstractActivity {
22  
23  	private final ShellView shellView;
24  	private ClientFactory clientFactory;
25  	private final UserRestService userService;
26  	private final AppPlaceHistoryMapper historyMapper;
27  	
28  	public ShellActivity(ShellPlace place, ClientFactory clientFactory) {
29  		super(clientFactory);
30  		
31  		GWT.log("Creating 'ShellActivity'");
32  		
33  		this.shellView = clientFactory.getShellView();
34  		this.clientFactory = clientFactory;
35  		this.userService = clientFactory.getUserService();
36  		
37  		((RestServiceProxy)userService).setResource(new Resource(""));
38  		
39  		historyMapper = GWT.create(AppPlaceHistoryMapper.class);
40  		setLoggedInUser();
41  	}
42  	
43  	@Override
44  	public void start(AcceptsOneWidget container, EventBus eventBus) {
45  		container.setWidget(shellView);
46  	}
47  	
48  	private void setLoggedInUser() {
49  		if (clientFactory.getLoggedInUser() == null) {
50  			ShellActivity.this.shellView.setLoggedInUser(null);
51  			this.shellView.setLogoutToken(historyMapper.getToken(new LoginPlace(ConstsUtil.LOGIN)));
52  			return;
53  		}
54  		
55  		this.shellView.setLogoutToken(historyMapper.getToken(new LoginPlace(ConstsUtil.LOGOUT)));
56  		if (clientFactory.getLoggedInUser().getEmail() != null 
57  				&& clientFactory.getLoggedInUser().getEmail().trim().length() > 0) {
58  			
59  			ShellActivity.this.shellView.setLoggedInUser(clientFactory.getLoggedInUser());
60  			return;
61  		}
62  		
63  		userService.getUser(ConstsUtil.DEFAULT_AUTH_HEADER, clientFactory.getLoggedInUser().getId(), new MethodCallback<UserDto>() {
64  			public void onFailure(Method method, Throwable caught) {
65  				GWT.log("ShellActivity: Error loading data");
66  				Window.alert(getConstants().errorLoadingTags());
67  			}
68  			public void onSuccess(Method method, UserDto user) {
69  				GWT.log("ShellActivity: Loading User: " + user.getEmail());
70  				
71  				// now that we have the full user info, update the user logged in user in the client factory
72  				clientFactory.setLoggedInUser(user);
73  				ShellActivity.this.shellView.setLoggedInUser(user);
74  			}
75  		});
76  	}
77  }
