20  import com.google.gwt.user.client.ui.InlineLabel;
21  import com.google.gwt.user.client.ui.SimplePanel;
22  import com.google.gwt.user.client.ui.Widget;
23  
24  public class ShellViewImplDesktop extends Composite implements ShellView {
25  
26      private ClientFactory clientFactory;
27  
28      private AppConstants constants;
29  
30      private static MyDockLayoutPanelUiBinder uiBinder = GWT.create(MyDockLayoutPanelUiBinder.class);
31  
32      interface MyDockLayoutPanelUiBinder extends UiBinder<Widget, ShellViewImplDesktop> {
33      }
34  
35      @UiField
36      SimplePanel navigationPanel;
37  
38      @UiField
39      CustomScrollPanel scrollPanel;
40  
41      @UiField
42      SimplePanel contentPanel;
43  
44      @UiField
45      InlineLabel loggedInUser;
46  
47      @UiField
48      InlineHyperlink loginLogoutLink;
49  
50      public ShellViewImplDesktop(ClientFactory clientFactory) {
51          GWT.log("Creating 'ShellViewImplDesktop'");
52  
53          this.clientFactory = clientFactory;
54          this.constants = clientFactory.getConstants();
55  
56          initWidget(uiBinder.createAndBindUi(this));
57  
58          this.loginLogoutLink.setVisible(false);
59          this.scrollPanel.removeHorizontalScrollbar();
60  
61          // Start ActivityManager for the main widget with our ActivityMapper
62          ActivityMapper navigationActivityMapper = new NavigationActivityMapper(
63                  clientFactory);
64          ActivityManager navigationActivityManager = new ActivityManager(
65                  navigationActivityMapper, this.clientFactory.getEventBus());
66          navigationActivityManager.setDisplay(navigationPanel);
67  
68          // right side
69          ActivityMapper contentActivityMapper = new ContentActivityMapper(clientFactory);
70          ActivityManager contentActivityManager = new ActivityManager(
71                  contentActivityMapper, this.clientFactory.getEventBus());
72  
73          // contentPanel.getElement().getStyle().setOverflow(Overflow.HIDDEN);
74          contentPanel.getElement().getStyle().setOverflow(Overflow.AUTO);
75          contentActivityManager.setDisplay(contentPanel);
76      }
77  
78      @Override
79      public void setLogoutToken(String token) {
80          loginLogoutLink.setTargetHistoryToken(token);
81      }
82  
83      @Override
84      public void setLoggedInUser(UserDto user) {
85          if (user == null) {
86              loginLogoutLink.setText(constants.login());
87              loggedInUser.setText("");
88          } else {
89              loginLogoutLink.setText(constants.logout());
90              loggedInUser.setText(user.getEmail() + " | ");
91          }
92      }
93  }
