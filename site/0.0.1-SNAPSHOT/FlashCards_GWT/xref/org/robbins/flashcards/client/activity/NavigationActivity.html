17  package org.robbins.flashcards.client.activity;
18  
19  import org.robbins.flashcards.client.factory.ClientFactory;
20  import org.robbins.flashcards.client.mvp.AppPlaceHistoryMapper;
21  import org.robbins.flashcards.client.place.ListFlashCardsPlace;
22  import org.robbins.flashcards.client.place.ListTagsPlace;
23  import org.robbins.flashcards.client.place.NewFlashCardPlace;
24  import org.robbins.flashcards.client.place.NewTagPlace;
25  import org.robbins.flashcards.client.ui.NavigationView;
26  import org.robbins.flashcards.util.ConstsUtil;
27  
28  import com.google.gwt.activity.shared.AbstractActivity;
29  import com.google.gwt.core.client.GWT;
30  import com.google.gwt.event.shared.EventBus;
31  import com.google.gwt.user.client.ui.AcceptsOneWidget;
32  
33  public class NavigationActivity extends AbstractActivity {
34  
35      private final NavigationView navigationView;
36  
37      public NavigationActivity(ClientFactory clientFactory) {
38          GWT.log("Creating 'NavigationActivity'");
39  
40          this.navigationView = clientFactory.getNavigationView();
41  
42          bind();
43      }
44  
45      @Override
46      public void start(AcceptsOneWidget panel, EventBus eventBus) {
47          panel.setWidget(navigationView.asWidget());
48      }
49  
50      public void bind() {
51  
52          AppPlaceHistoryMapper historyMapper = GWT.create(AppPlaceHistoryMapper.class);
53  
54          navigationView.setNewFlashCardToken(historyMapper.getToken(new NewFlashCardPlace(
55                  ConstsUtil.FLASHCARD_FORM)));
56          navigationView.setListFlashCardsToken(historyMapper.getToken(new ListFlashCardsPlace(
57                  ConstsUtil.LIST_FLASHCARDS)));
58          navigationView.setNewTagToken(historyMapper.getToken(new NewTagPlace(
59                  ConstsUtil.TAG_FORM)));
60          navigationView.setListTagsToken(historyMapper.getToken(new ListTagsPlace(
61                  ConstsUtil.LIST_TAGS)));
62      }
63  }
