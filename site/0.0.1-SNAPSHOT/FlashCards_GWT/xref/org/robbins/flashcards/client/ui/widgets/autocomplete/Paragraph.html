2   package org.robbins.flashcards.client.ui.widgets.autocomplete;
3   
4   import com.google.gwt.user.client.DOM;
5   import com.google.gwt.user.client.ui.HasText;
6   import com.google.gwt.user.client.ui.Widget;
7   
8   /**
9    * This widget is to create
10   * <p>
11   * elements in a page.
12   */
13  public class Paragraph extends Widget implements HasText {
14  
15      public Paragraph() {
16          setElement(DOM.createElement("p"));
17      }
18  
19      public Paragraph(String text) {
20          this();
21          setText(text);
22      }
23  
24      @Override
25      public String getText() {
26          return getElement().getInnerText();
27      }
28  
29      @Override
30      public void setText(String text) {
31          getElement().setInnerText(text);
32      }
33  }
