2   package org.robbins.flashcards.client.ui.widgets;
3   
4   import com.google.gwt.cell.client.AbstractCell;
5   import com.google.gwt.cell.client.ValueUpdater;
6   import com.google.gwt.core.client.GWT;
7   import com.google.gwt.dom.client.Element;
8   import com.google.gwt.dom.client.EventTarget;
9   import com.google.gwt.dom.client.NativeEvent;
10  import com.google.gwt.safehtml.client.SafeHtmlTemplates;
11  import com.google.gwt.safehtml.shared.SafeHtml;
12  import com.google.gwt.safehtml.shared.SafeHtmlBuilder;
13  import com.google.gwt.safehtml.shared.SafeHtmlUtils;
14  
15  public class LinkCell extends AbstractCell<String> {
16  
17      /**
18       * The HTML templates used to render the cell.
19       */
20      interface Templates extends SafeHtmlTemplates {
21  
22          /**
23           * The template for this Cell, which includes styles and a value.
24           *
25           * @param styles the styles to include in the style attribute of the div
26           * @param value the safe value. Since the value type is {@link SafeHtml}, it will
27           *        not be escaped before including it in the template. Alternatively, you
28           *        could make the value type String, in which case the value would be
29           *        escaped.
30           * @return a {@link SafeHtml} instance
31           */
32          @SafeHtmlTemplates.Template("<a href=\"javascript:;\">{0}</a>")
33          SafeHtml anchor(SafeHtml value);
34      }
35  
36      /**
37       * Create a singleton instance of the templates used to render the cell.
38       */
39      private static Templates templates = GWT.create(Templates.class);
40  
41      public LinkCell() {
42          super("click", "keydown");
43      }
44  
45      @Override
46      public void render(com.google.gwt.cell.client.Cell.Context context, String value,
47              SafeHtmlBuilder sb) {
48          /*
49           * Always do a null check on the value. Cell widgets can pass null to cells if the
50           * underlying data contains a null, or if the data arrives out of order.
51           */
52          if (value == null) {
53              return;
54          }
55  
56          // If the value comes from the user, we escape it to avoid XSS attacks.
57          SafeHtml safeValue = SafeHtmlUtils.fromString(value);
58  
59          // Use the template to create the Cell's html.
60          SafeHtml rendered = templates.anchor(safeValue);
61          sb.append(rendered);
62      }
63  
64      /**
65       * Called when an event occurs in a rendered instance of this Cell. The parent element
66       * refers to the element that contains the rendered cell, NOT to the outermost element
67       * that the Cell rendered.
68       */
69      @Override
70      public void onBrowserEvent(Context context, Element parent, String value,
71              NativeEvent event, ValueUpdater<String> valueUpdater) {
72          // Let AbstractCell handle the keydown event.
73          super.onBrowserEvent(context, parent, value, event, valueUpdater);
74  
75          // Handle the click event.
76          if ("click".equals(event.getType())) {
77              // Ignore clicks that occur outside of the outermost element.
78              EventTarget eventTarget = event.getEventTarget();
79              if (parent.getFirstChildElement().isOrHasChild(Element.as(eventTarget))) {
80                  GWT.log("LinkCell: Cell clicked");
81                  doAction(value, valueUpdater);
82              }
83          }
84      }
85  
86      /**
87       * onEnterKeyDown is called when the user presses the ENTER key will the Cell is
88       * selected. You are not required to override this method, but its a common convention
89       * that allows your cell to respond to key events.
90       */
91      @Override
92      protected void onEnterKeyDown(Context context, Element parent, String value,
93              NativeEvent event, ValueUpdater<String> valueUpdater) {
94          doAction(value, valueUpdater);
95      }
96  
97      private void doAction(String value, ValueUpdater<String> valueUpdater) {
98          // Trigger a value updater. In this case, the value doesn't actually
99          // change, but we use a ValueUpdater to let the app know that a value
100         // was clicked.
101         if (valueUpdater != null) {
102             valueUpdater.update(value);
103         }
104     }
105 }
