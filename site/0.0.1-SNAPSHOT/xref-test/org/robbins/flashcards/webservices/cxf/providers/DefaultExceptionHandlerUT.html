4   import static org.hamcrest.CoreMatchers.is;
5   import static org.junit.Assert.assertThat;

7   import javax.ws.rs.WebApplicationException;
8   import javax.ws.rs.core.Response;
9   import javax.ws.rs.core.Response.Status;
10  
11  import org.junit.Test;
12  import org.junit.experimental.categories.Category;
13  import org.robbins.flashcards.providers.DefaultExceptionHandler;
14  import org.robbins.flashcards.webservices.exceptions.GenericWebServiceException;
15  import org.robbins.flashcards.webservices.exceptions.WebServicesError;
16  import org.robbins.tests.UnitTest;
17  
18  @Category(UnitTest.class)
19  public class DefaultExceptionHandlerUT {
20  
21      private final DefaultExceptionHandler handler = new DefaultExceptionHandler();
22  
23      private GenericWebServiceException exception;
24  
25      @Test
26      public void toResponse_withUnexpectedException() {
27  
28          WebApplicationException webApplicationException = new WebApplicationException();
29  
30          Response result = handler.toResponse(webApplicationException);
31  
32          WebServicesError webServicesError = (WebServicesError) result.getEntity();
33  
34          assertThat(webServicesError, is(WebServicesError.class));
35          assertThat(webServicesError.getErrorId(),
36                  is(Status.INTERNAL_SERVER_ERROR.getStatusCode()));
37      }
38  
39      @Test
40      public void toResponse_withUnexpectedExceptionAndMessage() {
41  
42          Throwable throwable = new Throwable("error_message");
43          WebApplicationException webApplicationException = new WebApplicationException(
44                  throwable);
45  
46          Response result = handler.toResponse(webApplicationException);
47  
48          WebServicesError webServicesError = (WebServicesError) result.getEntity();
49  
50          assertThat(webServicesError, is(WebServicesError.class));
51          assertThat(webServicesError.getErrorId(),
52                  is(Status.INTERNAL_SERVER_ERROR.getStatusCode()));
53      }
54  
55      @Test
56      public void toResponse_withExpectedException() {
57          exception = new GenericWebServiceException(Status.NOT_FOUND,
58                  Status.NOT_FOUND.getReasonPhrase());
59  
60          Response result = handler.toResponse(exception);
61  
62          WebServicesError webServicesError = (WebServicesError) result.getEntity();
63  
64          assertThat(webServicesError, is(WebServicesError.class));
65          assertThat(webServicesError.getErrorId(), is(Status.NOT_FOUND.getStatusCode()));
66      }
67  }
