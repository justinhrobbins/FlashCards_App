5   import java.util.HashSet;
6   import java.util.List;

10  import javax.inject.Inject;
11  import javax.ws.rs.core.Context;
12  import javax.ws.rs.core.MediaType;
13  import javax.ws.rs.core.Response;
14  import javax.ws.rs.core.Response.ResponseBuilder;
15  import javax.ws.rs.core.UriInfo;

17  import org.apache.commons.lang3.StringUtils;
18  import org.apache.cxf.jaxrs.ext.ResponseHandler;
19  import org.apache.cxf.jaxrs.model.OperationResourceInfo;
20  import org.apache.cxf.jaxrs.model.Parameter;
21  import org.apache.cxf.message.Message;
22  import org.robbins.flashcards.exceptions.ServiceException;
23  import org.robbins.flashcards.jackson.CustomObjectMapper;
24  import org.robbins.flashcards.service.util.FieldInitializerUtil;
25  import org.robbins.flashcards.webservices.exceptions.GenericWebServiceException;
26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  import org.springframework.stereotype.Component;
29  
30  import com.fasterxml.jackson.databind.ObjectWriter;
31  import com.fasterxml.jackson.databind.ser.impl.SimpleBeanPropertyFilter;
32  import com.fasterxml.jackson.databind.ser.impl.SimpleFilterProvider;
33  
34  @Component("partialResponseFilter")
35  public class PartialResponseFilter implements ResponseHandler {
36  
37      private static final Logger LOGGER = LoggerFactory.getLogger(PartialResponseFilter.class);
38  
39      @Inject
40      private FieldInitializerUtil fieldInitializer;
41  
42      @Inject
43      private CustomObjectMapper objectMapper;
44  
45      @Context
46      private UriInfo uriInfo;
47  
48      @Override
49      public Response handleResponse(Message m, OperationResourceInfo ori, Response response) {
50  
51          if (ori == null) {
52              return null;
53          }
54  
55          // exit now if not an http GET method
56          if (!StringUtils.equals(ori.getHttpMethod(), "GET")) {
57              return null;
58          }
59  
60          // exit now if not a 200 response, no sense in apply filtering if not a
61          // '200 OK'
62          if (response.getStatus() != Response.Status.OK.getStatusCode()) {
63              return null;
64          }
65  
66          // exit now if we are not returning json
67          if (!ori.getProduceTypes().get(0).toString().equals(MediaType.APPLICATION_JSON)) {
68              return null;
69          }
70  
71          // get a reference to the response entity. the entity holds the payload
72          // of our response
73          Object entity = response.getEntity();
74  
75          // try to get the 'fields' parameter from the QueryString
76          String fields = uriInfo.getQueryParameters().getFirst("fields");
77  
78          // if the 'fields' QueryString is blank, then check to see if we have
79          // @DefaultValue for 'fields'
80          if (StringUtils.isBlank(fields)) {
81              // get the Parameters for this Resource
82              List<Parameter> parameters = ori.getParameters();
83              for (Parameter parm : parameters) {
84                  // is the current Parameter named 'fields'?
85                  if (StringUtils.equals(parm.getName(), "fields")) {
86                      // get the default value for 'fields'
87                      fields = parm.getDefaultValue();
88  
89                      // now that we found 'fields', there's no need to keep
90                      // looping
91                      break;
92                  }
93              }
94              // If 'fields' is still blank then we don't have a value from either
95              // the QueryString or @DefaultValue
96              if (StringUtils.isBlank(fields)) {
97                  LOGGER.debug("Did not find 'fields' pararm for Resource '"
98                          + uriInfo.getPath() + "'");
99                  return null;
100             }
101         }
102 
103         Set<String> filterProperties = this.getFieldsAsSet(fields);
104 
105         try {
106             // is the entity a collection?
107             if (entity instanceof Collection<?>) {
108                 fieldInitializer.initializeEntity((Collection<?>) entity,
109                         filterProperties);
110             } // is the entity an array?
111             else if (entity instanceof Object[]) {
112                 fieldInitializer.initializeEntity((Object[]) entity, filterProperties);
113             } else {
114                 fieldInitializer.initializeEntity(entity, filterProperties);
115             }
116         } catch (ServiceException e) {
117             throw new GenericWebServiceException(Response.Status.INTERNAL_SERVER_ERROR, e);
118         }
119 
120         // apply the Jackson filter and return the Response
121         return applyFieldsFilter(filterProperties, entity, response);
122     }
123 
124     // configure the Jackson filter for the speicified 'fields'
125     private Response applyFieldsFilter(Set<String> filterProperties, Object object,
126             Response response) {
127         SimpleFilterProvider filterProvider = new SimpleFilterProvider();
128         filterProvider.addFilter("apiFilter",
129                 SimpleBeanPropertyFilter.filterOutAllExcept(filterProperties));
130         filterProvider.setFailOnUnknownId(false);
131 
132         return applyWriter(object, filterProvider, response);
133     }
134 
135     // Get a JSON string from Jackson using the provided filter.
136     // Note we are using the ObjectWriter rather than the ObjectMapper directly.
137     // According to the Jackson docs,
138     // "ObjectWriter instances are immutable and thread-safe: they are created by ObjectMapper"
139     // You should not change config settings directly on the ObjectMapper while
140     // using it (changing config of ObjectMapper is not thread-safe
141     private Response applyWriter(Object object, SimpleFilterProvider filterProvider,
142             Response response) {
143 
144         try {
145             ObjectWriter writer = objectMapper.writer(filterProvider);
146             String jsonString = writer.writeValueAsString(object);
147 
148             // replace the Response entity with our filtered JSON string
149             ResponseBuilder builder = Response.fromResponse(response);
150             builder = builder.entity(jsonString);
151             return builder.build();
152         } catch (Exception e) {
153             LOGGER.error(e.getMessage(), e);
154             throw new GenericWebServiceException(Response.Status.INTERNAL_SERVER_ERROR, e);
155         }
156     }
157 
158     // Convert the vectorized 'fields' parameter to a Set<String>
159     private Set<String> getFieldsAsSet(String fields) {
160         Set<String> filterProperties = new HashSet<String>();
161         StringTokenizer st = new StringTokenizer(fields, ",");
162         while (st.hasMoreTokens()) {
163             String field = st.nextToken();
164 
165             // never allow 'userpassword' to be passed even if it was
166             // specifically requested
167             if (field.equals("userpassword")) {
168                 continue;
169             }
170 
171             // add the field to the Set<String>
172             filterProperties.add(field);
173         }
174         return filterProperties;
175     }
