10  import org.robbins.flashcards.client.factory.ClientFactory;
11  import org.robbins.flashcards.client.place.EditFlashCardPlace;
12  import org.robbins.flashcards.client.place.EditTagPlace;

25  import com.google.gwt.core.client.GWT;
26  import com.google.gwt.event.shared.EventBus;
27  import com.google.gwt.place.shared.PlaceController;
28  import com.google.gwt.user.client.Window;
29  import com.google.gwt.user.client.ui.AcceptsOneWidget;
30  
31  public class FlashCardsActivity extends AppAbstractActivity {
32  
33      private List<FlashCardDto> flashCards;
34  
35      private final FlashCardRestService flashCardService;
36  
37      private final EventBus eventBus;
38  
39      private final FlashCardsView display;
40  
41      private PlaceController placeController;
42  
43      public FlashCardsActivity(ClientFactory clientFactory) {
44          super(clientFactory);
45  
46          GWT.log("Creating 'FlashCardsActivity'");
47  
48          this.flashCardService = clientFactory.getFlashCardService();
49          this.eventBus = clientFactory.getEventBus();
50          this.display = clientFactory.getFlashCardsView();
51          this.placeController = clientFactory.getPlaceController();
52  
53          ((RestServiceProxy) flashCardService).setResource(new Resource(""));
54  
55          getRegistrations().add(
56                  LoadFlashCardEvent.register(this.eventBus,
57                          new LoadFlashCardEventHandler() {
58  
59                              @Override
60                              public void onLoadFlashCard(LoadFlashCardEvent event) {
61                                  GWT.log("FlashCardsActivity: 'Load FlashCard' event");
62                                  FlashCardsActivity.this.placeController.goTo(new EditFlashCardPlace(
63                                          Long.toString(event.getFlashCardId())));
64                              }
65                          }));
66  
67          getRegistrations().add(
68                  LoadTagEvent.register(this.eventBus, new LoadTagEventHandler() {
69  
70                      @Override
71                      public void onLoadTag(LoadTagEvent event) {
72                          GWT.log("TagsActivity: 'Load Tag' event");
73                          FlashCardsActivity.this.placeController.goTo(new EditTagPlace(
74                                  Long.toString(event.getTagId())));
75                      }
76                  }));
77  
78          getRegistrations().add(
79                  DeleteFlashCardEvent.register(this.eventBus,
80                          new DeleteFlashCardEventHandler() {
81  
82                              @Override
83                              public void onDeleteFlashcard(DeleteFlashCardEvent event) {
84                                  GWT.log("FlashCardsActivity: 'Delete FlashCard' event");
85  
86                                  flashCardService.deleteFlashCards(
87                                          ConstsUtil.DEFAULT_AUTH_HEADER,
88                                          event.getFlashCardId(),
89                                          new MethodCallback<java.lang.Void>() {
90  
91                                              @Override
92                                              public void onFailure(Method method,
93                                                      Throwable caught) {
94                                                  GWT.log("DeleteTagActivity: Error deleting data");
95                                                  Window.alert(getConstants().errorDeletingTag());
96                                              }
97  
98                                              @Override
99                                              public void onSuccess(Method method,
100                                                     java.lang.Void result) {
101                                                 GWT.log("DeleteTagActivity: Tag Deleted");
102                                                 FlashCardsActivity.this.placeController.goTo(new ListFlashCardsPlace(
103                                                         ""));
104                                             }
105                                         });
106                             }
107                         }));
108     }
109 
110     @Override
111     public void start(AcceptsOneWidget container, EventBus eventBus) {
112         container.setWidget(display.asWidget());
113 
114         String fields = ConstsUtil.DEFAULT_FLASHCARDS_LIST_FIELDS;
115         fetchFlashCardDetails(fields);
116     }
117 
118     private void fetchFlashCardDetails(String fields) {
119 
120         // load the table with data
121         flashCardService.getFlashCards(ConstsUtil.DEFAULT_AUTH_HEADER, fields,
122                 new MethodCallback<List<FlashCardDto>>() {
123 
124                     @Override
125                     public void onFailure(Method method, Throwable caught) {
126                         GWT.log("FlashCardsActivity: Error loading data");
127                         Window.alert(getConstants().errorLoadingFlashCard());
128                     }
129 
130                     @Override
131                     public void onSuccess(Method method, List<FlashCardDto> result) {
132                         flashCards = result;
133                         GWT.log("FlashCardsActivity: Loading FlashCard list: "
134                                 + flashCards.size() + " flashCards");
135 
136                         // flashCardCellTable.setInput(flashCards);
137                         display.setData(flashCards);
138                     }
139                 });
140     }
141 }
