21  public class TagListCell extends AbstractCell<List<TagDto>> {
22  
23      /**
24       * The HTML templates used to render the cell.
25       */
26      interface Templates extends SafeHtmlTemplates {
27  
28          /**
29           * The template for this Cell, which includes styles and a value.
30           *
31           * @param styles the styles to include in the style attribute of the div
32           * @param value the safe value. Since the value type is {@link SafeHtml}, it will
33           *        not be escaped before including it in the template. Alternatively, you
34           *        could make the value type String, in which case the value would be
35           *        escaped.
36           * @return a {@link SafeHtml} instance
37           */
38          @SafeHtmlTemplates.Template("<div name=\"{0}\" style=\"{1}\"><a href=\"javascript:;\">{2}</a></div>")
39          SafeHtml cell(String name, SafeStyles styles, String value);
40      }
41  
42      /**
43       * Create a singleton instance of the templates used to render the cell.
44       */
45      private static Templates templates = GWT.create(Templates.class);
46  
47      public TagListCell() {
48          super("click", "keydown");
49      }
50  
51      @Override
52      public void render(com.google.gwt.cell.client.Cell.Context context,
53              List<TagDto> tagList, SafeHtmlBuilder sb) {
54          /*
55           * Always do a null check on the value. Cell widgets can pass null to cells if the
56           * underlying data contains a null, or if the data arrives out of order.
57           */
58  
59          if (tagList == null) {
60              return;
61          }
62  
63          // generate the cell
64          // SafeStyles tagStyle = SafeStylesUtils.fromTrustedString("float:left;");
65          SafeStyles tagStyle = SafeStylesUtils.fromTrustedString("background: none repeat scroll 0 0 #DEE7F8; border: 1px solid #CAD8F3; border-radius: 5px 5px 5px 5px; color: #555555; cursor: pointer; float: left; font-size: 11px; margin: 2px; padding: 0 5px;");
66          String toRender = "";
67  
68          for (TagDto tag : tagList) {
69              /*
70               * if (toRender.length() > 0){ toRender +=
71               * "<div style=\"float:left\">,&nbsp;</div>"; }
72               */
73              toRender += templates.cell(Long.toString(tag.getId()), tagStyle,
74                      tag.getName()).asString();
75          }
76          sb.appendHtmlConstant(toRender);
77      }
78  
79      /**
80       * Called when an event occurs in a rendered instance of this Cell. The parent element
81       * refers to the element that contains the rendered cell, NOT to the outermost element
82       * that the Cell rendered.
83       */
84      @Override
85      public void onBrowserEvent(Context context, Element parent, List<TagDto> tagList,
86              NativeEvent event, ValueUpdater<List<TagDto>> valueUpdater) {
87          // Let AbstractCell handle the keydown event.
88          super.onBrowserEvent(context, parent, tagList, event, valueUpdater);
89  
90          // Handle the click event.
91          if ("click".equals(event.getType())) {
92              // Ignore clicks that occur outside of the outermost element.
93              EventTarget eventTarget = event.getEventTarget();
94  
95              if (parent.isOrHasChild(Element.as(eventTarget))) {
96                  // use this to get the selected element!!
97                  Element el = Element.as(eventTarget);
98  
99                  // check if we really click on the image
100                 if (el.getParentElement().getNodeName().equalsIgnoreCase("DIV")) {
101                     doAction(el.getParentElement().getAttribute("name"), valueUpdater);
102                 }
103             }
104         }
105     }
106 
107     private void doAction(String value, ValueUpdater<List<TagDto>> valueUpdater) {
108         GWT.log("Cell clicked: " + value);
109 
110         // Trigger a value updater. In this case, the value doesn't actually
111         // change, but we use a ValueUpdater to let the app know that a value
112         // was clicked.
113         TagDto tag = new TagDto(Long.parseLong(value));
114 
115         ArrayList<TagDto> tagList = new ArrayList<TagDto>();
116         tagList.add(tag);
117 
118         if (valueUpdater != null) {
119             valueUpdater.update(tagList);
120         }
121     }
122 }
