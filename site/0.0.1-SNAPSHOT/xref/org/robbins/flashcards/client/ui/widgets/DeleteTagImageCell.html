2   package org.robbins.flashcards.client.ui.widgets;
3   
4   import org.robbins.flashcards.client.ui.images.Images;
5   import org.robbins.flashcards.model.TagDto;
6   
7   import com.google.gwt.cell.client.AbstractCell;
8   import com.google.gwt.core.client.GWT;
9   import com.google.gwt.safehtml.client.SafeHtmlTemplates;
10  import com.google.gwt.safehtml.shared.SafeHtml;
11  import com.google.gwt.safehtml.shared.SafeHtmlBuilder;
12  import com.google.gwt.safehtml.shared.SafeHtmlUtils;
13  import com.google.gwt.user.client.ui.AbstractImagePrototype;
14  
15  public class DeleteTagImageCell extends AbstractCell<TagDto> {
16  
17      /**
18       * The HTML templates used to render the cell.
19       */
20      interface Templates extends SafeHtmlTemplates {
21  
22          /**
23           * The template for this Cell, which includes styles and a value.
24           *
25           * @param styles the styles to include in the style attribute of the div
26           * @param value the safe value. Since the value type is {@link SafeHtml}, it will
27           *        not be escaped before including it in the template. Alternatively, you
28           *        could make the value type String, in which case the value would be
29           *        escaped.
30           * @return a {@link SafeHtml} instance
31           */
32          @SafeHtmlTemplates.Template("<a href=\"javascript:;\">{0}</a>")
33          SafeHtml cell(SafeHtml value);
34      }
35  
36      /**
37       * Create a singleton instance of the templates used to render the cell.
38       */
39      private static Templates templates = GWT.create(Templates.class);
40  
41      private Images images = (Images) GWT.create(Images.class);
42  
43      public DeleteTagImageCell() {
44          super("click");
45      }
46  
47      @Override
48      public void render(Context context, TagDto tag, SafeHtmlBuilder sb) {
49          if (tag != null) {
50              if ((tag.getFlashcards() != null) && (tag.getFlashcards().size() > 0)) {
51                  sb.append(SafeHtmlUtils.fromTrustedString(AbstractImagePrototype.create(
52                          images.deleteDisabled()).getHTML()));
53              } else {
54                  SafeHtml html = SafeHtmlUtils.fromTrustedString(AbstractImagePrototype.create(
55                          images.delete()).getHTML());
56                  sb.append(templates.cell(html));
57              }
58          }
59      }
60  }
