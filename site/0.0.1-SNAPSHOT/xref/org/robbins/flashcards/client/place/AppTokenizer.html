2   package org.robbins.flashcards.client.place;
3   
4   import org.robbins.flashcards.util.ConstsUtil;
5   
6   import com.google.gwt.core.client.GWT;
7   import com.google.gwt.place.shared.Place;
8   import com.google.gwt.place.shared.PlaceTokenizer;
9   import com.google.gwt.place.shared.Prefix;
10  
11  @Prefix("")
12  public class AppTokenizer implements PlaceTokenizer<Place> {
13  
14      @Override
15      public Place getPlace(String token) {
16          GWT.log("AppTokenizer - Token: " + token);
17  
18          if (token.equals(ConstsUtil.TAG_FORM)) {
19              return new NewTagPlace(token);
20  
21          } else if (token.equals(ConstsUtil.LIST_TAGS)) {
22              return new ListTagsPlace(token);
23  
24          } else if (token.equals(ConstsUtil.FLASHCARD_FORM)) {
25              return new NewFlashCardPlace(token);
26  
27          } else if (token.equals(ConstsUtil.LIST_FLASHCARDS)) {
28              return new ListFlashCardsPlace(token);
29  
30          } else if (token.equals(ConstsUtil.LOGIN)) {
31              return new LoginPlace(token);
32  
33          } else if (token.equals(ConstsUtil.LOGOUT)) {
34              return new LogoutPlace(token);
35          } else {
36              // couldn't determine place
37              GWT.log("Could not determine which Place to call");
38              return null;
39          }
40      }
41  
42      @Override
43      public String getToken(Place place) {
44  
45          if (place instanceof NewTagPlace) {
46              return ((NewTagPlace) place).getPlaceName();
47  
48          } else if (place instanceof ListTagsPlace) {
49              return ((ListTagsPlace) place).getPlaceName();
50  
51          } else if (place instanceof NewFlashCardPlace) {
52              return ((NewFlashCardPlace) place).getPlaceName();
53  
54          } else if (place instanceof ListFlashCardsPlace) {
55              return ((ListFlashCardsPlace) place).getPlaceName();
56  
57          } else if (place instanceof LoginPlace) {
58              return ((LoginPlace) place).getPlaceName();
59  
60          } else if (place instanceof LogoutPlace) {
61              return ((LogoutPlace) place).getPlaceName();
62  
63          } else {
64              // couldn't find an activity
65              return null;
66          }
67      }
68  
69  }
