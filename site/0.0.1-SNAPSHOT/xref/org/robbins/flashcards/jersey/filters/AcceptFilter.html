19  /**
20   * Overrides the 'accept' HTTP header with value parsed from the querystring 'accept'
21   * parameter. The 'accept' parameter must match
22   * 
23   * @author Justin Robbins
24   * 
25   */
26  @Component("jerseyAcceptFilter")
27  public class AcceptFilter implements ContainerRequestFilter {
28  
29      static final Logger logger = LoggerFactory.getLogger(AcceptFilter.class);
30  
31      private static String ACCEPT = "accept";
32  
33      @Override
34      public ContainerRequest filter(ContainerRequest request) {
35          logger.debug("AcceptFilter");
36  
37          MultivaluedMap<String, String> queryParametersMap = request.getQueryParameters();
38  
39          if (queryParametersMap.containsKey(ACCEPT)) {
40              List<String> queryparmsList = queryParametersMap.get(ACCEPT);
41              logger.debug(queryparmsList.toString());
42  
43              // does the 'accept' header match either json or xml?
44              if ((!queryparmsList.contains(MediaType.APPLICATION_JSON))
45                      && (!queryparmsList.contains(MediaType.APPLICATION_XML))) {
46                  throw new GenericWebServiceException(Response.Status.BAD_REQUEST,
47                          "'accept' parameter must container either "
48                                  + MediaType.APPLICATION_JSON + " or "
49                                  + MediaType.APPLICATION_XML);
50              }
51  
52              MultivaluedMap<String, String> headers = request.getRequestHeaders();
53              headers.put(HttpHeaders.ACCEPT, queryparmsList);
54          }
55  
56          return request;
57      }
58  }
