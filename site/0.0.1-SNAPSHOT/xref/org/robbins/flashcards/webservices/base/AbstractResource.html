1   package org.robbins.flashcards.webservices.base;
2   
3   import java.util.HashSet;
4   import java.util.Set;
5   import java.util.StringTokenizer;
6   
7   import org.apache.cxf.common.util.StringUtils;
8   
9   public abstract class AbstractResource {
10  	
11  	// Convert the vectorized 'fields' parameter to a Set<String>
12  	protected Set<String> getFieldsAsSet(String fields) {
13  		
14  		if (StringUtils.isEmpty(fields)) {
15  			return null;
16  		}
17  		
18  		Set<String> filterProperties = new HashSet<String>();
19  		StringTokenizer st = new StringTokenizer(fields, ",");
20  		while (st.hasMoreTokens()) {
21  			String field = st.nextToken();
22  
23  			// never allow 'userpassword' to be passed even if it was
24  			// specifically requested
25  			if (field.equals("userpassword")) {
26  				continue;
27  			}
28  
29  			// add the field to the Set<String>
30  			filterProperties.add(field);
31  		}
32  		return filterProperties;
33  	}
34  }
