3   import javax.inject.Inject;
4   import javax.ws.rs.Consumes;
5   import javax.ws.rs.GET;
6   import javax.ws.rs.PUT;
7   import javax.ws.rs.Path;
8   import javax.ws.rs.PathParam;
9   import javax.ws.rs.Produces;
10  import javax.ws.rs.QueryParam;
11  import javax.ws.rs.core.Response;
12  
13  import org.robbins.flashcards.dto.TagDto;
14  import org.robbins.flashcards.exceptions.ServiceException;
15  import org.robbins.flashcards.facade.TagFacade;
16  import org.robbins.flashcards.facade.base.GenericCrudFacade;
17  import org.robbins.flashcards.webservices.base.AbstractGenericResource;
18  import org.robbins.flashcards.webservices.exceptions.GenericWebServiceException;
19  import org.springframework.stereotype.Component;
20  
21  import com.wordnik.swagger.annotations.Api;
22  import com.wordnik.swagger.annotations.ApiOperation;
23  
24  @Path("/v1/tags/")
25  @Component("tagsResource")
26  @Api(value="/v1/tags", description = "Operations about Tags")
27  @Produces({"application/xml", "application/json"})
28  @Consumes({"application/xml","application/json"})
29  public class TagsResource extends AbstractGenericResource<TagDto, Long> {
30  
31  	@Inject
32  	private TagFacade tagFacade;
33  	
34  	protected GenericCrudFacade<TagDto> getFacade() {
35  		return tagFacade;
36  	}
37  
38  	@GET
39  	@Path("/search")
40  	@ApiOperation(value = "Find Tag by Name", response = TagDto.class)
41  	public TagDto searchByName(@QueryParam("name") String name) {
42  		
43  		TagDto tagDto = tagFacade.findByName(name);
44  
45  		if (tagDto == null) {
46  			throw new GenericWebServiceException(Response.Status.NOT_FOUND,
47  					"Entity not found: " + name);
48  		}
49  		return tagDto;
50  	}
51  	
52  	@Override
53  	@PUT
54  	@Path("/{id}")
55  	@ApiOperation(value = "Replace a Tag")
56  	public Response put(@PathParam("id") Long id, TagDto dto) {
57  
58  		if (dto.getCreatedBy() == null) {
59  			TagDto orig;
60  			try {
61  				orig = tagFacade.findOne(id);
62  			} catch (ServiceException e) {
63  				throw new GenericWebServiceException(
64  						Response.Status.INTERNAL_SERVER_ERROR, e);
65  			}
66  			dto.setFlashcards(orig.getFlashcards());
67  			dto.setCreatedBy(orig.getCreatedBy());
68  			dto.setCreatedDate(orig.getCreatedDate());
69  		}
70  		return super.put(id,  dto);
71  	}
72  }
