79      @Override
80      public void validate() {
81          if (flashCard.getQuestion().length() == 0) {
82              addFieldError("flashCard.question", getText("error.flashcard.question"));
83          }
84  
85          if (flashCard.getAnswer().length() == 0) {
86              addFieldError("flashCard.answer", getText("error.flashcard.answer"));
87          }
88  
89          // remove any empty Link fields from the Arry List
90          Iterator<String> iterator = this.flashCard.getLinks().iterator();
91          while (iterator.hasNext()) {
92              String next = iterator.next();
93              if (next.length() == 0) {
94                  iterator.remove();
95              }
96          }
97      }

99      public String saveOrUpdate() {
100         // workaround that converts that Tags passed by the form to Tag objects on the
101         // FlashCard
102         this.flashCard.setTags(convertToTags(getExplodedTags()));
103 
104         try {
105             if ((this.flashCard.getId() != null) && (this.flashCard.getId() != 0)) {
106                 FlashCard existingFlashCard = flashcardService.findOne(this.flashCard.getId());
107                 existingFlashCard.setQuestion(this.flashCard.getQuestion());
108                 existingFlashCard.setAnswer(this.flashCard.getAnswer());
109                 existingFlashCard.setTags(this.flashCard.getTags());
110                 existingFlashCard.setLinks(this.flashCard.getLinks());
111 
112                 flashcardService.save(existingFlashCard);
113 
114                 logger.debug("Flash Card updated successfully");
115                 this.addActionMessage(getText("actionmessage.flashcard.updated"));
116             } else {
117                 flashcardService.save(this.flashCard);
118 
119                 logger.debug("Flash Card created successfully");
120                 this.addActionMessage(getText("actionmessage.flashcard.created"));
121             }
122             return "success";
123         } catch (Exception e) {
124             logger.error("unable to create or update FlashCard", e);
125             return "error";
126         }
127     }

129     private Set<Tag> convertToTags(String explodedTags) {
130         Set<Tag> tags = new HashSet<Tag>();
131 
132         if (getExplodedTags() == null) {
133             return null;
134         }
135 
136         // create an array of strings with a space (",") as their delimiter
137         String tokens[] = explodedTags.split(",");
138         for (String token : tokens) {
139             if (token.length() > 0) {
140                 Tag tag = tagService.findByName(token);
141                 tags.add(tag);
142             }
143         }
144         return tags;
145     }

147     private String convertFromTags(Collection<Tag> tags, boolean addQuotes) {
148 
149         StringBuilder sb = new StringBuilder();
150         String prefix = "";
151 
152         for (Tag tempTag : tags) {
153             sb.append(prefix);
154             prefix = ", ";
155 
156             if (addQuotes) {
157                 sb.append("'" + tempTag.getName() + "'");
158             } else {
159                 sb.append(tempTag.getName());
160             }
161         }
162 
163         return sb.toString();
164     }

182     @SkipValidation
183     public String display() {
184         try {
185             if ((this.flashCard.getId() != null) && (this.flashCard.getId() != 0)) {
186                 this.flashCard = flashcardService.findOne(this.flashCard.getId());
187             } else if (this.flashCard.getQuestion() != null) {
188                 this.flashCard = flashcardService.findByQuestion(this.flashCard.getQuestion());
189             }
190             this.setExplodedTags(convertFromTags(this.flashCard.getTags(), false));
191 
192             // get a list of Tags from the data tier
193             this.tagList = null;
194 
195             tagList = tagService.findAll();
196 
197             this.setAllExplodedTags(convertFromTags(tagList, true));
198 
199             return "success";
200         } catch (Exception e) {
201             logger.error("Exception in display():", e);
202 
203             return "error";
204         }
205     }

207     @SkipValidation
208     public String browse() {
209         // reset the FlashCard list
210         this.flashCardList = null;
211 
212         try {
213 
214             // if there is a ViewType add it to the HTTP Session
215             if (this.getViewType() != null) {
216                 logger.debug("viewType found in Request: '" + this.getViewType() + "'");
217                 httpSession.put("viewType", this.getViewType());
218             }
219             // otherwise, try to get the ViewType from the HTTP Session
220             else if (httpSession.containsKey("viewType")) {
221                 logger.debug("viewType found in Session: '"
222                         + (String) httpSession.get("viewType") + "'");
223                 this.setViewType((String) httpSession.get("viewType"));
224             }
225 
226             // if we have specified Tags to limit the FlashCard results
227             if ((this.getExplodedTags() != null)
228                     && (!this.getExplodedTags().equals("All"))) {
229 
230                 logger.debug("ExplodedTags found in Request");
231 
232                 // add the Exploded Tags to the HTTP Session
233                 httpSession.put("explodedTags", this.getExplodedTags());
234 
235                 // workaround that converts that Tags passed by the form to Tag objects on
236                 // the FlashCard
237                 this.flashCard.setTags(convertToTags(getExplodedTags()));
238 
239                 // get a list of Flash Cards from the data tier
240                 this.flashCardList = flashcardService.findByTagsIn(this.flashCard.getTags());
241             }
242             // if we have the Tag filter saved in the HTTP Session
243             else if ((httpSession.containsKey("explodedTags"))
244                     && (this.getExplodedTags() == null)) {
245 
246                 logger.debug("ExplodedTags found in HTTP Session");
247 
248                 // do we have Tags saved in the session?
249                 this.setExplodedTags((String) httpSession.get("explodedTags"));
250 
251                 // workaround that converts that Tags passed by the form to Tag objects on
252                 // the FlashCard
253                 this.flashCard.setTags(convertToTags(getExplodedTags()));
254 
255                 // get a list of Flash Cards from the data tier
256                 this.flashCardList = flashcardService.findByTagsIn(this.flashCard.getTags());
257             }
258             // otherwise, let's grab all FlashCards for all Tags
259             else {
260                 logger.debug("Retrieving FlashCards for all Tags");
261 
262                 // retrieve all the FlashCards
263                 this.flashCardList = flashcardService.findAll();
264             }
265 
266             // get a list of Tags from the data tier
267             this.tagList = null;
268             tagList = tagService.findAll();
269 
270             return "success";
271         } catch (Exception e) {
272             logger.error("Exception in browse():", e);
273 
274             return "error";
275         }
