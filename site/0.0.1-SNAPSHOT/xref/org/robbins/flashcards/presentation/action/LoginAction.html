58      // the OpenID Selector form will submit to this Action method
59      public String validateOpenId() throws Exception {
60          // get rid of trailing slash
61          if (getOpenid_identifier().endsWith("/")) {
62              setOpenid_identifier(getOpenid_identifier().substring(0,
63                      getOpenid_identifier().length() - 1));
64          }
65  
66          LOGGER.debug("The requested OpenId identifier is: " + getOpenid_identifier());
67  
68          // determine a return_to URL where the application will receive
69          // the authentication responses from the OpenID provider
70          String returnUrl = getServerContext(request) + RETURN_ACTION;
71          LOGGER.debug("return URL: " + returnUrl);
72  
73          // construct the destination Url to send to the Open Id provider
74          String destinationUrl = OpenId4JavaAuthenticator.getValidateOpenIdUrl(returnUrl,
75                  this.getOpenid_identifier(), httpSession, application);
76  
77          // redirect to the Auth Request
78          response.sendRedirect(destinationUrl);
79  
80          // no need to return a view
81          return NONE;
82      }
83  
84      public String authenticateOpenId() throws Exception {
85          Map<String, String[]> parmList = request.getParameterMap();
86  
87          // extract the receiving URL from the HTTP request
88          final StringBuffer receivingURL = request.getRequestURL();
89          final String queryString = request.getQueryString();
90  
91          if (queryString != null && queryString.length() > 0) {
92              receivingURL.append("?").append(request.getQueryString());
93          }
94  
95          LOGGER.debug(receivingURL.toString().replaceAll("&", "\n"));
96  
97          // verify the user has authenticated with the Open Id provider and
98          // get a reference to the authenticated User
99          User authenticatingUser = OpenId4JavaAuthenticator.getAuthenticatedUser(parmList,
100                 receivingURL, httpSession, application);
101 
102         // does the user already exist?
103         User existingUser = userService.findUserByOpenid(authenticatingUser.getOpenid());
104         if (existingUser != null) {
105             authenticatingUser.setId(existingUser.getId());
106             authenticatingUser.setCreatedBy(existingUser.getCreatedBy());
107             authenticatingUser.setCreatedDate(existingUser.getCreatedDate());
108 
109             loggedInUser.setId(existingUser.getId());
110         } else {
111             // this is a new user and he doesn't have an id yet, temporarily set it to the
112             // service user (which is me)
113             loggedInUser.setId(1L);
114         }
115 
116         // save the user
117         User persistedUser = userService.save(authenticatingUser);
118 
119         httpSession.put("user", persistedUser);
120 
121         // get the id of the user. if he's new then this is our first opportunity to get
122         // his id
123         Long loggedInUserId = persistedUser.getId();
124 
125         // set the id on the loggedInUser which we'll use for auditing
126         loggedInUser.setId(loggedInUserId);
127 
128         // retrieve the original URL from the Session
129         String desitinationURL = (String) httpSession.get("originalURL");
130 
131         // was a destination URL provided?
132         if (desitinationURL == null) {
133             LOGGER.debug("No destination URL provided, will send to Home");
134             return "home";
135         } else {
136             LOGGER.debug("Redirecting to : " + desitinationURL);
137             response.sendRedirect(desitinationURL);
138             return NONE;
139         }
140     }

142     @SuppressWarnings("rawtypes")
143     public String logout() {
144         try {
145             // invalidate the user's session
146             httpSession.remove("user");
147 
148             if (httpSession instanceof org.apache.struts2.dispatcher.SessionMap) {
149                 try {
150                     ((org.apache.struts2.dispatcher.SessionMap) httpSession).invalidate();
151                 } catch (IllegalStateException e) {
152                     LOGGER.error("Exception in logout()", e);
153                 }
154             }
155 
156             return "success";
157         } catch (Exception e) {
158             LOGGER.error("Exception in logout():", e);
159 
160             return "error";
161         }
162     }

164     private String getServerContext(final HttpServletRequest request) {
165         // Get the base url.
166         final StringBuilder serverPath = new StringBuilder();
167 
168         serverPath.append(request.getScheme() + "://");
169         serverPath.append(request.getServerName());
170 
171         if (request.getServerPort() != HTTP_PORT) {
172             serverPath.append(":" + request.getServerPort());
173         }
174         serverPath.append(request.getContextPath());
175 
176         return serverPath.toString();
177     }
